    def customPrint(*args, **kwargs):
    sep = ' '
    end = '\n'
    res = ""
  
    if "sep" in kwargs:
        sep = kwargs.get("sep")
    if "end" in kwargs:
        end = kwargs.get("end")
      
    for i in args:
        res += str(i) + sep
      
    if len(args) > 0:
        l = list(res)
        l.pop()
        res = "".join(l)
    res += end
  
    return res
  
print(customPrint(1, 2, 3, sep='-', end='^^'))
  
print(customPrint(1, 2, 3))
